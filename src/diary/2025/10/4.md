```python
#读取
file='MarkedTown_01.txt'
with open(file,'r') as f :
    marks=[frozenset(map(int,each.split(','))) for each in f.read().split('\n') ]

#数据分析
types = frozenset(marks)
MAX_MARK = max(b for a in types for b in a)
MARKS_EACH_ONE = len(marks[0])
POPULATION=len(marks)
marks_count={x : marks.count(x) for x in types }
number_count = {}
def count():
    number_count.clear()
    for mark in marks_count:
        for num in mark:
            number_count[num] = number_count.get(num, 0) + marks_count[mark]
    print('统计完成')
count()
print(f'初始数据整理完成,最大编号{MAX_MARK},每人{MARKS_EACH_ONE}个编号，共{POPULATION}人')

#计算
day = 0
dead_nums = set()
while True:
    day += 1
    change = False
    for num in number_count:
        if num not in dead_nums and day >= number_count[num] + num:
            dead_nums.add(num)
            change = True
    change and print(f'Day{day} 已被确认:{dead_nums}')
    change = False
    marks_mirror = marks_count.copy()
    for mark in marks_count:
        if mark.issubset(dead_nums):
            print(f'Day{day} 编号{set(mark)}已被察觉，消失{marks_mirror.pop(mark,'ERROR！')}人')
            change = True
    marks_count = marks_mirror
    if not marks_count:
        break
    change and count()
print(f'结束了！第{day}天，白茫茫一片大地真干净')
```
